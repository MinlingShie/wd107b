請寫一個程式計算 10! ，也就是 10*9*8*....*1。


function factorial(n){
   var i=1,r=1;
   while(i<=n){
     r=r*i
     i=i+1
   }
   return r;
}

var f=factorial(10)
console.log('10!=',f)







請寫一個程式印出九九乘法表。


for(i=2;i<=9;i++){
   for(j=2;j<=9;j++){
     console.log(i,'*',j,'=',i*j);
   }
}
